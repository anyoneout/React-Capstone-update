var socket = new WebSocket('ws://localhost:8080'); socket.onmessage = function (event) { var elementId = event.data.trim(); // Ensure the ID matches exactly var element = document.getElementById(elementId); if (element) { // Highlight the element element.style.outline = '3px solid red'; // Remove the highlight after 2 seconds setTimeout(function () { element.style.outline = ''; }, 2000); } else { console.warn('Element with ID "' + elementId + '" not found.'); } };
